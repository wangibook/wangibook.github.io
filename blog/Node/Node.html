# 四、fs文件系统

# 读文件：fs.readFile(path,可选参数,回调){}

function fsRead(path) {
  return new Promise(function(resolve,reject){
    fs.readFile(path,{encoding: 'utf-8'},function(err,data) {
      if(data) {
        // 成功执行
        resolve(data)
      } else {
        // 失败执行
        reject(err)
      }
    })
  })
}

# 写文件：fs.writeFile(path,content,回调){}

function fsWrite(path,content){
  return new Promise(function(resolve,reject){
    fs.writeFile(path,content,function(err){
      if(err) {
        console.log('写入出错')
        reject(err)
      } else {
        console.log('写入成功')
      }
    })
  })
}

function readFileFn(title) {
  return new Promise(function(resolve,reject){

      if(err){
        reject(err)
      } else {

      }
    })
  })
}

# 七、爬虫总结

# （1）目标：确定想要获取的数据

  • 确定想要的数据在什么页面上（一般详细的数据在详情页）
  • 确定哪些页面可以链接到这些页面（一般分类列表页面会有详情页的链接数据）
  • 寻找页面之间和数据之间的规律

# （2）分析页

  • 获取数据的方式（正则，cherrio）
  • 分析数据是通过ajax请求，还是html自带的数据
  • 如果是ajax请求的数据，那么需要获取ajax请求的链接，一般会请求到json格式数据
  • 如果数据在html里面，那么就用cherrio通过选择器将内容选中
